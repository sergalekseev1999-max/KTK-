window.validateRequiredFields = function() {

            var customDisplay = document.getElementById(el.id + '_display');

            var realTarget = customDisplay || targetEl;

                var dataList = (el.id === 'supplierSelect') ? (globalData.suppList||[]) : (globalData.termListFull||[]);

                    }

            }

        }

    };

window.searchContainers = function() {

      var raw = document.getElementById('searchMassInput');

      if (!raw) return;

      var text = raw.value.trim();

      if (!text) { window.showToast('Введите ID или ссылки для поиска!', 'error'); return; }

      

      // Normalize: split by newlines/commas/spaces, clean each token, rejoin with newline

      var tokens = text.split(/[\r\n,;\s]+/).map(function(t){ return t.trim(); }).filter(function(t){ return t.length > 0; });

      var cleanText = tokens.join('\n');

      

      var container = document.getElementById('searchListContainer');

      if(container) container.innerHTML = '<div class="text-[#F5F5DC] animate-pulse text-xs text-center py-4">🔍 Поиск контейнеров (' + tokens.length + ' шт.)...</div>';

      if(massBtn) massBtn.classList.add('hidden');

      if(countTitle) countTitle.classList.add('hidden');

            .withSuccessHandler(function(data) {

              searchResults = Array.isArray(data) ? data : [];

            })

              var errMsg = String(err);

              if(container) container.innerHTML = '<div class="text-red-400 text-xs text-center py-4">❌ Ошибка поиска: ' + errMsg + '</div>';

              window.showToast('Ошибка поиска: ' + errMsg, 'error');

            })

            .searchInventoryMass(cleanText);

      } else {

          if(container) container.innerHTML = '<div class="text-yellow-400 text-xs text-center py-4">⚠️ Нет подключения к Google Apps Script</div>';
